class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        count = {}
        left = 0
        maxfreq = 0
        answer = 0
        for right in range(len(s)):
            char = s[right]
            count[char] = count.get(char, 0) + 1
            if count[char] > maxfreq:
                maxfreq = count[char]
            while (right - left + 1) - maxfreq > k:
                left_char = s[left]
                count[left_char] -= 1
                left += 1
            windowlen = right - left + 1
            if windowlen > answer:
                answer = windowlen

        return answer
