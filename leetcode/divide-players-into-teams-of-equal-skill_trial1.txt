class Solution:
    def dividePlayers(self, skill: List[int]) -> int:
        skill.sort()
        i=0
        j=len(skill)-1
        target=skill[i]+skill[j]
        total=0
        while i<j:
            if target != skill[i]+skill[j]:
                return -1
            total+=skill[i]*skill[j]
            i+=1
            j-=1
        return total
