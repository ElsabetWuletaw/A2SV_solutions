class Solution:
    def check(self,weights,m,days):
        total,day=0,1
        for w in weights:
            total += w
            if total>m:
                total=w
                day+=1
        return day<=days
    def shipWithinDays(self, weights: List[int], days: int) -> int:
        l=max(weights)
        h=sum(weights)
        while l<=h:
            m=(l+h)//2
            if self.check(weights,m,days):
                h=m-1
            else:
                l=m+1
        return l
