class Solution:
    def longestSemiRepetitiveSubstring(self, s: str) -> int:
        i = 0
        pairs = 0
        result = 1
        for j in range(1, len(s)):
            if s[j] == s[j - 1]:
                pairs += 1
            while pairs > 1:
                if s[i] == s[i + 1]:
                    pairs -= 1
                i += 1
            result = max(result, j - i + 1)

        return result
